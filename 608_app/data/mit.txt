LF.CommentCount({
  replacer: function(element, count) {
    var formatSingular = element.getAttribute("data-lf-format-singular");
    var formatPlural = element.getAttribute("data-lf-format-plural");
    if (formatSingular && formatPlural) {
      if (count == 1) {
        element.innerHTML = formatSingular.replace("@count", count);
      } else {
        element.innerHTML = formatPlural.replace("@count", count);
      }
    } else {
      element.innerHTML = count;
    }

    var minComments = 1;
    // Get the minimum comments from the HTML attribute, if there is one.
    var htmlMinComments = element.getAttribute("data-lf-min-comments");
    if (htmlMinComments) {
      minComments = 1 * htmlMinComments;
    }

    var displayElement = element;
    // If we need to hide/show a parent element instead of the element itself,
    // we specify the jQuery .parent() selector in the attribute data-lf-display-parents.
    var displayParentSelector = element.getAttribute("data-lf-display-parents");
    if (displayParentSelector) {
      var displayElementParents = $(displayElement).parents(displayParentSelector);
      if (displayElementParents.length > 0)
        displayElement = displayElementParents[0];
    }

    var displayAltElement = null;
    var displayAltParentsSelector = element.getAttribute("data-lf-display-alt-parents");
    var displayAltSelector = element.getAttribute("data-lf-display-alt");
    if (displayAltParentsSelector && displayAltSelector) {
      displayAltParents = $(element).parents(displayAltParentsSelector);
      if (displayAltParents.length > 0) {
        var displayAltParent = displayAltParents[0];
        var displayAltElements = $(displayAltParent).find(displayAltSelector);
        if (displayAltElements.length > 0)
          displayAltElement = displayAltElements[0];
      }
    }

    // Hide the element if it doesn't have enough comments.
    if (count >= minComments) {
      displayElement.style.display = "";
      // Hide the alternate element.
      if (displayAltElement)
        displayAltElement.style.display = "none";
    } else {
      displayElement.style.display = "none";
      // Show the alternate element.
      if (displayAltElement)
        displayAltElement.style.display = "";
    }
  }
});
